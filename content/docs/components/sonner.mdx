## Usage

```tsx
import { toast } from "sonner";

toast("Event has been created.");
```

### Add the Toaster component

```tsx
import { Toaster } from "@soar-design/soar-react-components";

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <head />
      <body>
        <main>{children}</main>
        <Toaster />
      </body>
    </html>
  );
}
```
