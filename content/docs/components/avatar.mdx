title: Avatar

---

<ComponentPreview name="avatar-demo" />

## Usage

```tsx
import {
  Avatar,
  AvatarFallback,
  AvatarImage,
} from "@soar-design/soar-react-components";

<Avatar>
  <AvatarImage src="https://github.com/shadcn.png" />
  <AvatarFallback>CN</AvatarFallback>
</Avatar>;
```
