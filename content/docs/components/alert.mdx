## Usage

```tsx
import { Alert, AlertDescription, AlertTitle } from "@soar-design/soar-react-components"

<Alert variant="default | destructive">
  <Terminal />
  <AlertTitle>Heads up!</AlertTitle>
  <AlertDescription>
    You can add components and dependencies to your app using the cli.
  </AlertDescription>
</Alert>
```
